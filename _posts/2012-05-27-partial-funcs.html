When you need to pass a function name as a parameter to something else but the param also needs input, you can "hide" calls using partial funcs.

from functools import partial

int( '1010', base=2 )
> 10
int( '1011', base=2 )
> 11
int( 'FF', base=16 )
> 255

frombinary = partial( int, base=2 )
fromhex    = partial( int, base=16 )

frombinary( '1010101' )
> 85
fromhex('FFFFFF')
> 16777215
fromhex('cafebabe')
> 3405691582L
